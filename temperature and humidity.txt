Temperature=int(input('Temperature = '))
Humidity=int(input('Humidity = '))
if(Temperature>=60):
    {
        print('Temperature is high')
    }
else:
    {
        print('Temperature is normal')
    }
if(Humidity>=1000):
    {
        print('Humidity is high')
    }
else:
    {
        print('Humidity is normal')
    }
